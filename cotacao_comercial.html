Siga os passos abaixo para ativar o salvamento automático de cotações na planilha. Feito uma vez, funciona para todos os vendedores.

1Abra a planilha e vá em Extensões → Apps Script
2Apague o código existente e cole o código abaixo
// Cole este código no Apps Script da sua planilha function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const now = new Date(); if (data.action === 'oc') { // Salva Ordem de Coleta em aba separada let shOC = ss.getSheetByName('Ordens de Coleta'); if (!shOC) { shOC = ss.insertSheet('Ordens de Coleta'); shOC.appendRow(['Nº OC','Data','Solicitante','Cotação','Local Coleta','Local Entrega','Remetente','CNPJ Rem','Destinatário','CNPJ Dest','Placa','Veículo','Motorista']); shOC.setFrozenRows(1); shOC.getRange(1,1,1,13).setBackground('#F5B800').setFontWeight('bold').setFontColor('#111111'); shOC.setColumnWidths(1,13,130); } const dateOC = Utilities.formatDate(now, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm'); shOC.appendRow([data.numero||'',dateOC,data.solicitante||'',data.cotacao||'',data.endColeta||'',data.endEntrega||'',data.remetente||'',data.cnpjRem||'',data.destinatario||'',data.cnpjDest||'',data.placa||'',data.veiculo||'',data.motorista||'']); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } else if (data.action === 'update') { // Atualiza Status e/ou Observação de uma cotação existente const sh = ss.getSheets()[0]; const allData = sh.getDataRange().getValues(); const headers = allData[0].map(String); const numCol = headers.indexOf('Numero'); const statusCol = headers.indexOf('Status'); const obsCol = headers.indexOf('Observacao'); if (numCol === -1) return ContentService.createTextOutput(JSON.stringify({success:false, error:'Col Numero not found'})).setMimeType(ContentService.MimeType.JSON); // Procura pela linha com o número da cotação for (let i = 1; i < allData.length; i++) { if (String(allData[i][numCol]) === String(data.numero)) { if (statusCol !== -1 && data.status !== undefined) sh.getRange(i+1, statusCol+1).setValue(data.status); if (obsCol !== -1 && data.obs !== undefined) sh.getRange(i+1, obsCol+1).setValue(data.obs); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({success:false, error:'Row not found'})).setMimeType(ContentService.MimeType.JSON); } else if (data.action === 'refazer') { // Atualiza cotação existente — limite de 2 revisões const sh = ss.getSheets()[0]; const allData = sh.getDataRange().getValues(); const headers = allData[0].map(String); const numCol = headers.indexOf('Numero'); if (numCol === -1) return ContentService.createTextOutput(JSON.stringify({success:false,error:'Col Numero not found'})).setMimeType(ContentService.MimeType.JSON); // Garante coluna Revisoes if (!headers.includes('Revisoes')) { const lastCol = sh.getLastColumn() + 1; sh.getRange(1, lastCol).setValue('Revisoes'); sh.getRange(1, lastCol).setBackground('#F5B800').setFontWeight('bold').setFontColor('#111111'); sh.setColumnWidth(lastCol, 100); headers.push('Revisoes'); } const revCol = headers.indexOf('Revisoes'); for (let i = 1; i < allData.length; i++) { if (String(allData[i][numCol]) === String(data.numero)) { const currentEdits = parseInt(allData[i][revCol] || '0', 10); if (currentEdits >= 2) return ContentService.createTextOutput(JSON.stringify({success:false,error:'Limite de revisoes atingido'})).setMimeType(ContentService.MimeType.JSON); const date = Utilities.formatDate(now, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm'); const updateMap = {'Data':date,'Vendedor':data.vendedor||'','Cliente':data.cliente||'','Modalidade':data.modalidade||'','Cidade Origem':data.origem||'','Cidade Destino':data.destino||'','Piso':data.piso||'','Valor':data.valor||'','Empresa':data.empresa||'','CNPJ Emit':data.cnpjEmit||'','CNPJ Tomador':data.cnpjTomador||''}; headers.forEach((h, colIdx) => { if (updateMap[h] !== undefined) sh.getRange(i+1, colIdx+1).setValue(updateMap[h]); }); sh.getRange(i+1, revCol+1).setValue(currentEdits + 1); return ContentService.createTextOutput(JSON.stringify({success:true,num:data.numero})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({success:false,error:'Row not found'})).setMimeType(ContentService.MimeType.JSON); } else { // Salva Cotação nova na primeira aba const sh = ss.getSheets()[0]; const date = Utilities.formatDate(now, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm'); const HEADERS = ['Data','Numero','Vendedor','Cliente','Modalidade','Cidade Origem','Cidade Destino','Piso','Valor','Status','Empresa','CNPJ Emit','CNPJ Tomador','Revisoes','Observacao']; if (sh.getLastRow() === 0) { sh.appendRow(HEADERS); sh.setFrozenRows(1); sh.getRange(1,1,1,HEADERS.length).setBackground('#F5B800').setFontWeight('bold').setFontColor('#111111'); sh.setColumnWidths(1,HEADERS.length,130); } else { const ex = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(String); if (!ex.includes('Cliente')) { const v=ex.indexOf('Vendedor'); const c=v!==-1?v+2:4; sh.insertColumnAfter(c-1); sh.getRange(1,c).setValue('Cliente').setBackground('#F5B800').setFontWeight('bold').setFontColor('#111111'); sh.setColumnWidth(c,130); } const ex2 = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(String); if (!ex2.includes('Revisoes')) { const lc=sh.getLastColumn()+1; sh.getRange(1,lc).setValue('Revisoes').setBackground('#F5B800').setFontWeight('bold').setFontColor('#111111'); sh.setColumnWidth(lc,100); } } const currentHeaders = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(String); const num = sh.getLastRow(); const rowMap = {'Data':date,'Numero':num,'Vendedor':data.vendedor||'','Cliente':data.cliente||'','Modalidade':data.modalidade||'','Cidade Origem':data.origem||'','Cidade Destino':data.destino||'','Piso':data.piso||'','Valor':data.valor||'','Status':'Pendente','Empresa':data.empresa||'','CNPJ Emit':data.cnpjEmit||'','CNPJ Tomador':data.cnpjTomador||'','Revisoes':0,'Observacao':''}; const newRow = currentHeaders.map(h => rowMap[h] !== undefined ? rowMap[h] : ''); sh.appendRow(newRow); return ContentService.createTextOutput(JSON.stringify({success:true,num})).setMimeType(ContentService.MimeType.JSON); } } catch(err) { return ContentService.createTextOutput(JSON.stringify({success:false, error:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheets()[0]; const data = sh.getDataRange().getValues(); if (data.length <= 1) return ContentService.createTextOutput(JSON.stringify({status:'ok',rows:[]})).setMimeType(ContentService.MimeType.JSON); const headers = data[0].map(String); const rows = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] === '' ? '' : String(row[i]); }); return obj; }).filter(r => Object.values(r).some(v => v)); return ContentService.createTextOutput(JSON.stringify({status:'ok', rows})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error', rows:[], error:err.toString()})).setMimeType(ContentService.MimeType.JSON); } }
